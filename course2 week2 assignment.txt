pollutant_mean <- function(directory, pollutant, ID)
	{
	 correctids <- newcompletedata[newcompletedata$ID == ID, ]

		{correctids$nitrate}
		else
		{correctids$sulfate}
	
	 complete_cases <- complete.cases(pollutant_column)
	
	 mean(pollutant_column[complete_cases])
